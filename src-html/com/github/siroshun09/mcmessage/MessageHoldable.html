017package com.github.siroshun09.mcmessage;
018
019import com.github.siroshun09.mcmessage.message.KeyedMessage;
020import com.github.siroshun09.mcmessage.message.Message;
021import org.jetbrains.annotations.NotNull;
022import org.jetbrains.annotations.Nullable;
023import org.jetbrains.annotations.Unmodifiable;
024
025import java.util.Objects;
026import java.util.Set;
027
028public interface MessageHoldable {
029
030    @Nullable Message getMessage(@NotNull String key);
031
032    default @NotNull Message getMessage(@NotNull String key, @NotNull String def) {
033        Message message = getMessage(key);
034        return message != null ? message : Message.create(def);
035    }
036
037    default @NotNull Message getMessage(@NotNull KeyedMessage keyedMessage) {
038        Objects.requireNonNull(keyedMessage);
039        return getMessage(keyedMessage.getKey(), keyedMessage.getMessage());
040    }
041
042    @NotNull @Unmodifiable Set<? extends KeyedMessage> getMessages();
043}
