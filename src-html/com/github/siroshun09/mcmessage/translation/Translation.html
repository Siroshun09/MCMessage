017package com.github.siroshun09.mcmessage.translation;
018
019import com.github.siroshun09.mcmessage.message.KeyedMessage;
020import com.github.siroshun09.mcmessage.message.TranslatedMessage;
021import com.github.siroshun09.mcmessage.MessageHoldable;
022import org.jetbrains.annotations.Contract;
023import org.jetbrains.annotations.NotNull;
024import org.jetbrains.annotations.Nullable;
025import org.jetbrains.annotations.Unmodifiable;
026
027import java.util.Locale;
028import java.util.Map;
029import java.util.Objects;
030import java.util.Set;
031import java.util.stream.Collectors;
032
033public interface Translation extends MessageHoldable {
034
035    @Contract(value = "_, _ -> new", pure = true)
036    static @NotNull Translation create(@NotNull Map<String, TranslatedMessage> messageMap, @NotNull Locale locale) {
037        return new TranslationImpl(messageMap, locale);
038    }
039
040    @Contract("_, _ -> new")
041    static @NotNull Translation create(@NotNull Set<TranslatedMessage> messageSet, @NotNull Locale locale) {
042        return new TranslationImpl(
043                messageSet.stream().collect(
044                        Collectors.toUnmodifiableMap(
045                                TranslatedMessage::getKey,
046                                t -> t
047                        )
048                ),
049                locale
050        );
051    }
052
053    @Override
054    @Nullable TranslatedMessage getMessage(@NotNull String key);
055
056    @Override
057    default @NotNull TranslatedMessage getMessage(@NotNull String key, @NotNull String def) {
058        var message = getMessage(key);
059        return message != null ? message : TranslatedMessage.create(key, def, getLocale());
060    }
061
062    @Override
063    default @NotNull TranslatedMessage getMessage(@NotNull KeyedMessage keyedMessage) {
064        Objects.requireNonNull(keyedMessage);
065        return getMessage(keyedMessage.getKey(), keyedMessage.getMessage());
066    }
067
068    @Override
069    @NotNull @Unmodifiable Set<TranslatedMessage> getMessages();
070
071    @NotNull Locale getLocale();
072}
